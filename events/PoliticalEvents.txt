# Major Factory-worker Strike

country_event = {
	
	id = 5001

	trigger = {
		government_in_exile = no
		or = {
			civil_law = open_society
			civil_law = limited_restrictions
		}
		not = { taxation_law = negligible_taxes }
		not = { popularity = 50 }
		not = { has_country_modifier = "factory_workers_demand_met" }
		any_owned_province = {
			industry = 5
		}
	}

	mean_time_to_happen =  {
		months = 36

		modifier = { factor = 2.0 war = yes }
		modifier = { factor = 2.0 has_country_flag = "great_war" }
		
		modifier = { factor = 1.5 organisation = 30 }
		modifier = { factor = 1.5 organisation = 50 }
		
		modifier = { factor = 0.9 dissent = 1 }
		modifier = { factor = 0.9 dissent = 2 }
		modifier = { factor = 0.9 dissent = 3 }
		modifier = { factor = 0.9 dissent = 4 }
		modifier = { factor = 0.9 dissent = 5 }
		modifier = { factor = 0.5 dissent = 10 }
		modifier = { factor = 0.5 dissent = 15 }
		modifier = { factor = 0.5 dissent = 20 }
		
		modifier = { factor = 1.5 civil_law = limited_restrictions }			# Less likely if not Open Society.
		modifier = { factor = 0.8 economic_law = war_economy }					# More likely if using harsh economic laws.
		modifier = { factor = 0.5 economic_law = total_economic_mobilisation }	# More likely if using harsh economic laws.
		modifier = { factor = 1.5 taxation_law = acceptable_taxes }				# Less likely if taxes are sensible.
		modifier = { factor = 0.8 taxation_law = high_taxes }					# More likely if taxes are high.
		modifier = { factor = 0.5 taxation_law = exploitive_taxes }				# More likely if taxes are high.
		modifier = {
			factor = 0.5
			war = no
			not = { economic_law = full_civilian_economy }						# Use low policies at peace!
		}
	}

	title = "Major worker strike"
	desc = "A major worker strike started in one of the nation's major factories. The unions demand higher wages and lessened workhours."
	picture = "major_worker_strike"

	option = {
		name = "Put them down!"			# It's hardly anything to worry about
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0.5
				dissent = 5
			}
			modifier = {
				factor = 0.25
				dissent = 10
			}
			modifier = {
				factor = 0.1
				dissent = 15
			}
		}
		random_list = {
			65 = { dissent = 5 popularity = -5 }		# People go home, but they won't forget.
			28 = { dissent = 8 popularity = -10 }		# People go home, but they are rather angry.
			7  = {
				dissent = 10 popularity = -15			# The situation escalates.
				random_owned = {
					limit = { industry = 5 }
					create_revolt = 1
				}
			}
		}
	}
	option = {
		name = "Let's have a deal!"
		ai_chance = {
			factor = 25
			modifier = {
				factor = 2.0
				dissent = 5
			}
			modifier = {
				factor = 3.0
				dissent = 10
			}
			modifier = {
				factor = 4.0
				dissent = 15
			}
		}
		popularity = 5
		organisation = -2
		dissent = -1
		add_country_modifier = {
			name = "factory_workers_demand_met"
			duration = -1
		}
	}
}
