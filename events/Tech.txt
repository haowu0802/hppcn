# Nuclear Research Lab captured!

country_event = {

	id = 7430
	
	trigger = {
		civil_nuclear_research = 2		# We have some advanced knowledge of Nucler Science
		any_owned_province = {
			has_building = "nuclear_reactor"
			controller = {
				war_with = THIS							# Our enemy captured our Nuclear Research Facility
				not = { civil_nuclear_research = 2 }	# and can learn something from it!
				civil_nuclear_research = 1
			}
		}
	}

	title = "Our Nuclear Research Lab was captured"
	desc = "A Nuclear Research Lab of ours was captured by the enemy! If we don't do something quick, they may be able to reverse-engineer some of our equipment there!"
	picture = "nuclear_test_lab"
	
	option = {
		name = "Try to sabotage the facility!"
		random_list = {
			50 = {
				any_owned = {
					limit = {
						controller = {
							war_with = THIS
							not = { civil_nuclear_research = 2 }
							civil_nuclear_research = 1
						}
						has_building = "nuclear_reactor"
					}
					controller = { country_event = 7439 }
				}
			}
			50 = {
				any_owned = {
					limit = {
						controller = {
							war_with = THIS
							not = { civil_nuclear_research = 2 }
							civil_nuclear_research = 1
						}
						has_building = "nuclear_reactor"
					}
					nuclear_reactor = -10
				}
			}
		}
	}

}
